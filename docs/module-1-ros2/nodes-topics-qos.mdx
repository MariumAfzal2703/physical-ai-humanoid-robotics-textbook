from collections import deque

class SlidingRate:
    def __init__(self, size: int = 100):
        self.times = deque(maxlen=size)

    def push(self, t: float) -> None:
        self.times.append(t)

    def hz(self) -> float:
        if len(self.times) < 2:
            return 0.0
        dt = self.times[-1] - self.times[0]
        return 0.0 if dt <= 0 else (len(self.times) - 1) / dt
