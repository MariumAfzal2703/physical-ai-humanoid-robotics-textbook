from dataclasses import dataclass

@dataclass
class Signal:
    infer_ms: float
    saturation: float
    fallback_hits: int


def severity(s: Signal) -> str:
    if s.fallback_hits > 0:
        return "critical"
    if s.infer_ms > 45 or s.saturation > 0.3:
        return "warning"
    return "ok"
