```python
import random
from dataclasses import dataclass

@dataclass
class DisturbancePlan:
    seed: int
    steps: int


def replayable_impulses(plan: DisturbancePlan) -> list[float]:
    rng = random.Random(plan.seed)
    return [round(rng.uniform(-0.06, 0.06), 4) for _ in range(plan.steps)]
```
